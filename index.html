Bit Destuffing Program

#include <stdio.h>
#include <string.h>
void main ()
{
    char os[60], fs[60]="";
    int i=0,j,k,count=0;
    printf("Enter the stuffed bitstring");
    scanf("%s",os);
    k=strlen(os)-8;
    // finding 11111 pattern & then stuffing with 0 //
    for(i=8,j=0;i<k;i++)
    {
        if(os[i]=='1')
        {
            count+=1;
            fs[j]='1';
            j++;
        }
        else if(os[i]=='0')
        {    
            fs[j]='0';
            j++;
            count=0;
        }
        if(count==5)
        {
            if(os[i+1]=='1')
            {
                printf("The string is invalid\n");
                break;
            }
            i++;
            count=0;
        }
    }
    printf("The string is %s \n",fs);
}
